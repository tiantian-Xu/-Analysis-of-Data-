from matplotlib import pyplot as plt
from wordcloud import WordCloud
# 词云显示
font = './FZYouHFW.TTF'           # 华文行楷
words = ",".join(new_data)
wc = WordCloud(font_path = font,                 #解决显示口字乱码问题
               background_color = 'white',        #背景颜色
               width = 1200, height =1000,     
               max_font_size = 250,
                max_words=30, 
               prefer_horizontal=15        #   调整词云中字体水平和垂直的多少
              ).generate(words)
plt.imshow(wc)   # 用plt显示图片
plt.axis('off')  # 不显示坐标轴
plt.savefig('商品评价词云.png')
plt.show()       # 显示图片
